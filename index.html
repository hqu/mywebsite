analyze.py
import matplotlib.pyplot as plt
import pandas as pd
from pathlib import Path

df = pd.read_csv(Path("datasaurus_all.csv"))
dino = df[df["shape"] == "dino"].copy()

fig, ax = plt.subplots(constrained_layout=True)
ax.scatter(dino["x"], dino["y"], s=36, c="#0f172a")
ax.set_title("Datasaurus Dino")
ax.set_xlabel("x")
ax.set_ylabel("y")
ax.set_aspect("equal")
fig.savefig("dino_scatter.png", dpi=200)
filter_dino.py
